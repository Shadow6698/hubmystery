-- Fun√ß√£o para adicionar bot√µes que executam scripts
local function AddScriptButtons(section, scripts)
    for _, script in ipairs(scripts) do
        local name, url = script[1], script[2]
        section:Button({
            Text = name,
            Callback = function()
                if url == "local" then
                    warn("Script "..name.." √© local, implemente voc√™ mesmo.")
                else
                    local success, err = pcall(function()
                        loadstring(game:HttpGet(url))()
                    end)
                    if not success then
                        warn("Erro ao executar "..name..": "..err)
                    end
                end
            end
        })
    end
end
